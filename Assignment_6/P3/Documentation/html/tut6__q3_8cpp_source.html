293 void linearfitline(double x[],double y[],int n)
294 {
295  int i;
296  double a,b;
297  double xsum=0,x2sum=0,ysum=0,xysum=0; //variables for sums/sigma of xi,yi,xi^2,xiyi etc
298  for (i=0;i<n;i++)
299  {
300  xsum=xsum+x[i]; //calculate sigma(xi)
301  ysum=ysum+y[i]; //calculate sigma(yi)
302  x2sum=x2sum+pow(x[i],2); //calculate sigma(x^2i)
303  xysum=xysum+x[i]*y[i]; //calculate sigma(xi*yi)
304  }
305  a=(n*xysum-xsum*ysum)/(n*x2sum-xsum*xsum); //calculate slope
306  b=(x2sum*ysum-xsum*xysum)/(x2sum*n-xsum*xsum); //calculate intercept
307  cout<<"\nThe linear fit:\n"<<a<<"x + "<<b<<endl;
308 }
